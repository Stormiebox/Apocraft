   /*************************Remove Paint From Light Bulbs*********************/

      Result:LightBulb,
      Time:250.0,
      Category:Paints,
      NeedToBeLearn:false,
      CanBeDoneFromFloor:true,
      Sound:MakePlaster,
   }
